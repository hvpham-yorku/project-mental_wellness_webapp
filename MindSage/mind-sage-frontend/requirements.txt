npm (v8 or higher)               # Node Package Manager
React (v18 or higher)            # JavaScript library for building user interfaces
React DOM (v18 or higher)        # React package for DOM rendering

rc-slider                       # Slider component for mood input
react-router-dom                # For routing in React applications (if applicable)
